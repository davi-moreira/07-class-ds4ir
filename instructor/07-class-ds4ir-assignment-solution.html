  1. Demonstre a evoluo da expectativa de vida no Brasil em 3 grficos: um com os dados completos, um com a decomposio da tendncia e um com a decomposio do resduo.
br <- gapminder %>% 
  filter(country == "Brazil") 

br %>% 
  ggplot(aes(year, lifeExp)) + 
  geom_line() + 
  ggtitle("Brasil - Completo")

m <- lm(lifeExp ~ year, data = br)

br %>% 
  add_predictions(m) %>%
  ggplot(aes(year, pred)) + 
  geom_line() + 
  ggtitle("Brasil - Tendncia")

br %>% 
  add_residuals(m) %>% 
  ggplot(aes(year, resid)) + 
  geom_hline(yintercept = 0, colour = "white", size = 3) + 
  geom_line() + 
  ggtitle("Brasil - Resduo")

by_country <- gapminder %>% 
  group_by(country, continent) %>% 
  nest()

country_model <- function(df) {
  lm(lifeExp ~ year, data = df)
}

by_country <- by_country %>% 
  mutate(model = map(data, country_model))

by_country <- by_country %>% 
  mutate(resids = map2(data, model, add_residuals))

resids <- unnest(by_country, resids)

glance <- by_country %>% 
  mutate(glance = map(model, broom::glance)) %>% 
  unnest(glance) %>% 
  filter(continent == "Asia") %>% 
  arrange(r.squared)
